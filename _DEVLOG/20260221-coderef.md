# single source of truth. example of markraw + upliftcode
```js
<script setup>
import { ref, onMounted, markRaw } from 'vue';
import Chart from 'chart.js/auto';

const chartCanvas = ref(null);
const chartInstance = ref(null);

onMounted(() => {
  const ctx = chartCanvas.value.getContext('2d');
  const chart = new Chart(ctx, {
    type: 'line',
    data: {
      labels: ['Jan', 'Feb', 'Mar'],
      datasets: [{
        label: 'Actual Sales',
        data: [10, 20, 15],
        borderColor: 'blue'
      }]
    }
  });

  // markRaw prevents Vue from tracking thousands of internal properties
  chartInstance.value = markRaw(chart);
});

const addProjectedLine = () => {
  if (!chartInstance.value) return;

  // 1. Mutate the raw object directly
  chartInstance.value.data.datasets.push({
    label: 'Projected Sales',
    data: [12, 25, 30],
    borderColor: 'red',
    borderDash: [5, 5] // Stylistic change for the new line
  });

  // 2. Trigger the Chart.js render engine manually
  chartInstance.value.update(); 
};
</script>

<template>
  <div>
    <button @click="addProjectedLine">Add Projected Line</button>
    <canvas ref="chartCanvas"></canvas>
  </div>
</template>
```

# add composable to prevent memory leak
- auto execute when unmount
```js
// useCleanup.js
import { onUnmounted } from 'vue'

export function useCleanup() {
  const cleanups = []
  
  // Register a function to be called on unmount
  const addCleanup = (fn) => cleanups.push(fn)
  
  onUnmounted(() => {
    cleanups.forEach(fn => fn())
  })
  
  return { addCleanup }
}

// In your component
<script setup>
import { onMounted } from 'vue'
import { useCleanup } from './useCleanup'

const { addCleanup } = useCleanup()

onMounted(() => {
  const handler = () => console.log('Resized')
  window.addEventListener('resize', handler)
  
  // Register cleanup automatically
  addCleanup(() => window.removeEventListener('resize', handler))
})
</script>
```
